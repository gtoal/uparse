begin
   comment 'LIBRARY' 1;
   real
     X,
     Y;
   integer
     CASE;
   procedure CHECK(B);
      value
        B;
      boolean
        B;
      if B then
        CASE ≔ CASE + 1
      else
      begin
         OUTREAL(99, X);
         OUTREAL(99, Y);
         OUTREAL(99, CASE)
      end;
   CASE ≔ 0;
   for X ≔ - 6·8,·001,
     13·4 do
        for Y ≔ - 6·8,·001,
        13·4 do
         begin
            if X = Y then
              CHECK(X ⩽ Y ∨ X ⩾ Y)
              else if X > Y then
              CHECK(X ⩾ Y)
              else if X < Y then
              CHECK(X ⩽ Y)
            else
              CHECK(false)
         end;
   OUTREAL(99, CASE);
end
