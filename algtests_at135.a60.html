begin
   comment 'LIBRARY' 1;
   real
     X,
     Y,
     Z,
     U;
   integer
     I,
     CASE;
   procedure COMPARE(U, V);
      value
        U,
        V;
      real
        U,
        V;
   begin
      CASE ≔ CASE + 1;
      if U ≠ V then
      begin
         OUTREAL(99, U);
         OUTREAL(99, V);
         OUTREAL(99, CASE)
      end;
   end;
   CASE ≔ 0;
   X ≔ 1·2;
   Y ≔ 16·96;
   Z ≔ - 13·4;
   for I ≔ 1 step 1 until 10 do
      begin
         COMPARE( - X × Y, - (X × Y));
         COMPARE( - X / Y, - (X / Y));
         COMPARE(X + Y + Z, (X + Y) + Z);
         COMPARE(X × Y × Z, (X × Y) × Z);
         COMPARE(X / Y / Z, (X / Y) / Z);
         COMPARE(X × Y / Z, (X × Y) / Z);
         COMPARE(X⭡2 / Y, (X × X) / Y);
         COMPARE(X × Y⭡2, X × (Y × Y));
         U ≔ 1·01 × X;
         X ≔ 1·01 × Y;
         Y ≔ 1·01 × Z;
         Z ≔ U;
      end;
   OUTREAL(99, 0)
end
