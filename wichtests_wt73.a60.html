begin
   comment LIBRARY 1;
   integer
     I,
     J,
     K,
     K1;
   real
     T1,
     T2;
   integerprocedure ACKERMANN(M, N);
      value
        M,
        N;
      integer
        M,
        N;
      ACKERMANN ≔         if M = 0 then
        N + 1
      else if N = 0 then
        ACKERMANN(M - 1, 1)
      else
        ACKERMANN(M - 1, ACKERMANN(M, N - 1));
   K ≔ 16;
   K1 ≔ 1;
   for I ≔ 1 step 1 until 6 do
      begin
         T1 ≔ CPUTIME;
         J ≔ ACKERMANN(3, I);
         T2 ≔ CPUTIME;
         if J ≠ K - 3 then
         begin
            OUTREAL(99, I);
            OUTREAL(99, J)
         end;
         OUTREAL(99, T2 - T1);
         OUTREAL(99, 3 × (T2 - T1) / (512 × K1 - 15 × K + 9 × I + 37) );
         K1 ≔ 4 × K1;
         K ≔ 2 × K;
      end
end
