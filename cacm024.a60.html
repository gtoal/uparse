real procedure TRIDAG(n, A, B, C, D); integer n ;
     array A, B, C, D ;
     comment ;
begin
   real w; integer j ;
   D[1] ≔ D[1]/B[1] ; w ≔ B[1] ;
   for j ≔ 2 step 1 until n do
      begin B[j - 1] ≔ C[j - 1]/w ; w ≔ B[j] - A[j - 1]× B[j - 1] ;
         D[j] ≔ (D[j] - A[j - 1] × D[j - 1])/w end ;
   for j ≔ 1 step 1 until n - 1 do
        D[n - j] ≔ D[n - j] - B[n - j] × D[n - j + 1]
end TRIDAG;
