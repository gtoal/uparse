begin
   integer
     I,
     J;
   for I ≔ - 2,
     - 1,
     1,
     2 do
        for J ≔ - 2 step 1 until 2 do
           if J ÷ I ≠ SIGN(J / I) × ENTIER(ABS(J / I))
           then
         begin
            OUTREAL(99, J);
            OUTREAL(99, I);
            OUTREAL(99, J ÷ I)
         end;
end
