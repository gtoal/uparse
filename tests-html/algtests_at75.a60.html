begin
   comment LIBRARY 9;
   integerarray
     EQ[1 : 3];
   integer
     I;
   real
     X,
     Y;
   procedure INC(A, B);
      value
        A,
        B;
      real
        A,
        B;
      if A < B then
        EQ[1] ≔ EQ[1] + 1
        else if A = B then
        EQ[2] ≔ EQ[2] + 1
      else
        EQ[3] ≔ EQ[3] + 1;
   for I ≔ 1 step 1 until 3 do
        EQ[I] ≔ 0;
   for I ≔ 2 step 1 until 100 do
      begin
         INREAL(98, X);
         Y ≔ 1 / (I × I - 1);
         OUTREAL(99, Y);
         INC(X, Y)
      end;
   for I ≔ 1 step 1 until 3 do
        OUTREAL(99, EQ[I]);
end
