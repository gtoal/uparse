begin
   comment LIBRARY 1;
   real
     I,
     J;
   array
     A[1 : 2, 1 : 3];
   real procedure GPS(I, N, Z, V);
      real
        I,
        N,
        Z,
        V;
   begin
      for I ≔ 1 step 1 until N do
           Z ≔ V;
      GPS ≔ 1
   end;
   OUTREAL(99, GPS(J, 3·0, I, GPS(I, 2·0, A[I, J], I + J)));
   for I ≔ 1 step 1 until 2 do
        for J ≔ 1 step 1 until 3 do
           if A[I, J] ≠ I + J then
           OUTREAL(99, 11 × I + J)
end
