   procedure movedisk(n, f, t);
      integer n;
        integer f;
        integer t;
   begin
      outstring (1, “move ”);
      outinteger(1, f);
      outstring (1, “ --> ”);
      outinteger(1, t);
      outstring (1, “\n”);
   end;

   procedure dohanoi(n, f, t, u);
      integer n;
        integer f;
        integer t;
        integer u;
   begin
      if n < 2 then
        movedisk(1, f, t)
      else
      begin
         dohanoi(n - 1, f, u, t);
         movedisk(1, f, t);
         dohanoi(n - 1, u, t, f);
      end;
   end;

   dohanoi(3, 1, 3, 2);
end
