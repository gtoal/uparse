begin
   integer procedure TSIGN(X);
      value
        X;
      real
        X;
      TSIGN ≔         if X > 0 then
        1
      else if X < 0 then
        - 1
      else
        0;
   integer procedure TENTIER(X);
      value
        X;
      real
        X;
      TENTIER ≔ X - ·5;
   real procedure TABS(X);
      value
        X;
      real
        X;
      TABS ≔         if X < 0 then
        - X
      else
        X;
   real
     X;
   for X ≔ - 100,
     - 100·1,
     - 99·4,
     - ·4,
     - ·1,
     0,·1,·4,·6,
     300 do
        if TSIGN(X) ≠ SIGN(X) ∨ TENTIER(X) ≠ ENTIER(X) ∨
        TABS(X) ≠ ABS(X) then
        OUTREAL(99, X)
end
