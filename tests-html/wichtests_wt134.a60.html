begin
   comment 'LIBRARY' 1;
   integer
     I,
     J,
     K;
   boolean
     P,
     T;
   procedure FAIL;
   begin
      OUTREAL(99, I);
      OUTREAL(99, J)
   end;
   T ≔ true;
   for I ≔ - 1,
     0,
     1 do
        for J ≔ - 1,
        0,
        1 do
         begin
            K ≔ 3 × (I + 1) + J + 2;
            if K ÷ 2 × 2 = K ≡ T then
              FAIL;
            T ≔ ¬ T;
            if (I + J > 1 ∧ J ≠ 1) ∨ (I + J < - 1∧ I ≠ - 1) then
              FAIL;
            if ¬ (K ⩾ 7 ∧ I ≠ 1 impl K < 0) then
              FAIL;
            P ≔ I = J;
            if P ∧ I × J < 0 then
              FAIL
         end;
   OUTREAL(99, 0)
end
