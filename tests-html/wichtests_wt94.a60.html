begin
   comment LIBRARY 1;
   integer
     J,
     K,
     S;
   integerprocedure F(N);
      value
        N;
      integer
        N;
   begin
      integer
        I,
        L;
      for I ≔ N,
        S ÷ 2 while S < 16 ∨ S = 28 do
         begin
            S ≔ S + I + N;
            L ≔ I
         end;
      F ≔ N + L;
   end F;
   S ≔ - 1;
   for J ≔ F(2),
     F(3) while S = 97 do
      begin
         S ≔ S + F(5) + J + S;
         K ≔ J
      end;
   S ≔ S + K;
   OUTREAL(99, S)
end
