begin
   integer i;
   real y;
   real array a[0:10];
   real procedure f(t); value t; real t; f ≔ sqrt(abs(t))+5×t⭡3;

   for i ≔ 0 step 1 until 10 do read(a[i]);
   for i ≔ 10 step -1 until 0 do begin
         y ≔ f(a[i]);
         if y > 400 then
           write(i, “TOO LARGE”)
         else
           write(i,y);
      end
end
